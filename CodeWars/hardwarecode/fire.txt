#include <ESP32Servo.h>

#define SERVO1_PIN 13  
#define SERVO2_PIN 12  
#define MQ_PIN 34     
#define FIRE_SIGNAL_PIN 32

Servo servo1; 
Servo servo2; 

void setup() {
  Serial.begin(115200);
  servo1.attach(SERVO1_PIN);
  servo2.attach(SERVO2_PIN);
  pinMode(FIRE_SIGNAL_PIN, INPUT); 
}

void loop() {
  int fireSignal = digitalRead(FIRE_SIGNAL_PIN);
  int sensorValue = analogRead(MQ_PIN);

  
  Serial.print("ðŸ”¥ MQ Sensor Value: ");
  Serial.println(sensorValue);

 
  if (fireSignal == HIGH || sensorValue > 3000) {
    Serial.println("ðŸš¨ Fire Detected! Stopping search & activating response!");

   
    servo1.write(90);
    delay(1000); 
   
    servo2.write(0);
    delay(500);
    servo2.write(180);
    delay(500);
    servo2.write(90); 

    return; 
  }

  
  for (int pos = 0; pos <= 180; pos += 2) {  
    servo1.write(pos);
    
    
    sensorValue = analogRead(MQ_PIN);
    Serial.print("ðŸ”¥ MQ Sensor Value: ");
    Serial.println(sensorValue);

    delay(100);
  }

  for (int pos = 180; pos >= 0; pos -= 2) {
    servo1.write(pos);

    
    sensorValue = analogRead(MQ_PIN);
    Serial.print("ðŸ”¥ MQ Sensor Value: ");
    Serial.println(sensorValue);

    delay(30);
  }
}
